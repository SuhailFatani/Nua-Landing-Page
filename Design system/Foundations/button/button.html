How to Use

1. Link CSS

<link rel="stylesheet" href="Design system/Foundations/button/button.css">

2. Basic buttons

<!-- Primary · lg --> <button class="nua-btn nua-btn--lg nua-btn--primary">Book a demo</button> <!-- Secondary color · md --> <button class="nua-btn nua-btn--md nua-btn--secondary-color">Learn more</button> <!-- Link · sm --> <a href="#" class="nua-btn nua-btn--sm nua-btn--link-color">View all</a>

3. With icons

<!-- Leading icon --> <button class="nua-btn nua-btn--lg nua-btn--primary"> <iconify-icon icon="solar:arrow-right-linear"></iconify-icon> Get started </button> <!-- Trailing icon --> <button class="nua-btn nua-btn--lg nua-btn--secondary-color"> Learn more <iconify-icon icon="solar:arrow-right-linear"></iconify-icon> </button> <!-- Icon only --> <button class="nua-btn nua-btn--lg nua-btn--primary nua-btn--icon-only" aria-label="Add"> <iconify-icon icon="solar:add-circle-linear"></iconify-icon> </button>

4. Destructive + Dot

<!-- Destructive --> <button class="nua-btn nua-btn--lg nua-btn--primary nua-btn--destructive">Delete</button> <!-- Dot indicator (add span inside any button) --> <button class="nua-btn nua-btn--lg nua-btn--secondary-color"> Alerts <span class="nua-btn__dot"></span> </button> <!-- Disabled --> <button class="nua-btn nua-btn--lg nua-btn--primary" disabled>Disabled</button>
