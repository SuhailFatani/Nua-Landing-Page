How to Use

1. Link CSS + Iconify

<link rel="stylesheet" href="Design system/Foundations/sidebar/sidebar.css"> <script src="https://cdn.jsdelivr.net/npm/iconify-icon@2/dist/iconify-icon.min.js"></script>

2. Default sidebar

<nav class="nua-sidebar"> <!-- Top: logo + nav --> <div class="nua-sidebar__top"> <a href="/" class="nua-sidebar__logo">...svg...</a> <div class="nua-sidebar__nav"> <a href="/dashboard" class="nua-sidebar__btn nua-sidebar__btn--active"> <span class="nua-sidebar__icon"> <iconify-icon icon="solar:widget-2-linear" class="icon-default"></iconify-icon> <iconify-icon icon="solar:widget-2-bold" class="icon-active"></iconify-icon> </span> <span class="nua-sidebar__text">Dashboard</span> </a> <!-- more buttons ... --> </div> </div> <!-- Bottom: settings button --> <a href="/settings" class="nua-sidebar__btn">...</a> </nav>

<nav class="nua-sidebar nua-sidebar--settings"> <span class="nua-sidebar__section-label">Settings</span> <div class="nua-sidebar__nav"> <a href="/settings/profile" class="nua-sidebar__btn nua-sidebar__btn--active"> <span class="nua-sidebar__icon"> <iconify-icon icon="solar:user-linear" class="icon-default"></iconify-icon> <iconify-icon icon="solar:user-bold" class="icon-active"></iconify-icon> </span> <span class="nua-sidebar__text">Profile</span> </a> </div> </nav>
