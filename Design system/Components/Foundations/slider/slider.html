How to Use
<!-- 1. Link the CSS --> <link rel="stylesheet" href="Design system/Foundations/slider/slider.css"> <!-- 2. HTML structure --> <div class="nua-slider"> <div class="nua-slider__header"> <div class="nua-slider__label">Label</div> <span class="nua-slider__value">8 hours</span> </div> <input type="range" class="nua-slider__input" min="1" max="8" value="8"> <div class="nua-slider__ticks"> <span>1</span>...<span>8</span> </div> <!-- optional error --> <div class="nua-slider__error"> <iconify-icon icon="solar:letter-linear"></iconify-icon> <span>Error message</span> </div> </div> <!-- 3. JS: live fill + value update --> <script> function updateSlider(inputId, valueId, fmt) { const el = document.getElementById(inputId); const pct = (el.value - el.min) / (el.max - el.min) * 100; el.style.setProperty('--_fill', pct + '%'); document.getElementById(valueId).textContent = fmt(el.value); } // Init all sliders on load document.querySelectorAll('.nua-slider__input').forEach(el => { const pct = (el.value - el.min) / (el.max - el.min) * 100; el.style.setProperty('--_fill', pct + '%'); }); </script>
